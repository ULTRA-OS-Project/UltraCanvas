\documentclass{article}
\usepackage{amsmath}
% The sign function as a piecewise definition (cases environment).
\begin{document}
\[
  \operatorname{sgn}(x) =
  \begin{cases}
    -1 & \text{if } x < 0, \\
    \phantom{-}0 & \text{if } x = 0, \\
    \phantom{-}1 & \text{if } x > 0.
  \end{cases}
\]
\end{document}
